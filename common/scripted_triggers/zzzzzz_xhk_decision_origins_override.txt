is_subterranean_empire = {
	OR = {
		has_origin = origin_subterranean
		has_origin = origin_subterranean_machines
		## ADDED THIS LINE
		has_country_flag = decision_origins_subterranean
	}
}

can_add_overtuned_traits = {
	hidden_trigger = { exists = from }
	from = { 
		has_overtuned_origin = yes
	}
}

can_remove_overtuned_traits = {
	hidden_trigger = { exists = from }
	from = {
		OR = {
			has_overtuned_origin = yes
			if = {
				limit = {
					OR = {
						has_biogenesis_dlc = no
						has_tradition = tr_genetics_adopt
					}
				}
				has_tradition = tr_genetics_resequencing
			}
			else = {
				has_tradition = tr_purity_genotype_regeneration
			}
		}
	}
}

can_be_leader = {
    OR = {
        has_citizenship_rights = yes
        has_citizenship_type = { type = citizenship_shroud_touched country = root }
    }
    # Rule out Traits for servitude & lack of free will
    can_think = yes
    NOT = { has_trait = trait_syncretic_proles }
    OR = {
        is_robot_pop_group = no
        root = {
            OR = {
                has_technology = tech_synthetic_leaders
                is_machine_empire = yes
                is_mechanical_empire = yes
                is_individual_machine = yes
                has_country_flag = synthetic_fertility_no_bodies_left
            }
        }
    }
    OR = {
        is_same_species = root
        root = {
            NAND = {
                has_necrophage_origin = yes
                has_trait = trait_necrophage
            }
        }
    }
}
